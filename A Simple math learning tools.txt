import java.util.Scanner;
public class ASimpleMathLearningTools {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        
        int num1 = (int) (Math.random() * 10);
        int num2 = (int) (Math.random() * 10);
        System.out.println("what is " + num1 + " + " + num2 + " = ?");
        
        int userResult = input.nextInt();
         int programResult = num1 + num2;
        
        if(userResult == programResult){
            System.out.println("Your are the master of mathematics");
           
        }else{
            System.out.println("You are not the master of mathematics!");
        }
        
    }
    
}
