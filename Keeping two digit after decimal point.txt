package keeping.two.digites.after.decimal.points;

/**
 *
 * @author H P
 */
import java.util.Scanner;
public class KeepingTwoDigitesAfterDecimalPoints {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        System.out.println("Enter product original price!!");
        double amount = input.nextFloat();
        System.out.println("Enter sale tax!");
        double tax = input.nextDouble();
        double taxAmount = (amount * tax ) / 100;
        System.out.printf("Product price after sale tax is %.2f" , (amount + taxAmount));
    }
    
}
