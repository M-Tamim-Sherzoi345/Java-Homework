import java.util.Scanner;
public class MaxOfTwoNumber {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        System.out.println("Enter Two number, I will teel you which one is max!");
        System.out.println("Enter Number first!");
        double num1 = input.nextDouble();
        System.out.println("Enter number second!");
        double num2 = input.nextDouble();
        if(num1 > num2){
            System.out.println("The first number is max from second number.");
        } else{
            System.out.println("The second number is max from first number.");
        }
        
    }
    
}
